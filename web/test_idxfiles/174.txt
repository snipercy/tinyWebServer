        Scrolled = window.pageYOffset;

       DoFixSub();
    }
    else

        DoReturnSub();
    }
    
    if (mapIsExpanded())

        DoReturnMap();
        return;
    }    
    
    if (Scrolled > FixedEle.MapY)
    {
        DoFixMap();
    }
    else
    {
        DoReturnMap();
    }
