function getCookie(Name){
	var search = Name + "=";
	if (document.cookie.length > 0){ 						// if there are any cookies
		offset = document.cookie.indexOf(search);
    	if (offset != -1){ 									// if cookie exists
      		offset += search.length	;	      				// set index of beginning of value
      		end = document.cookie.indexOf(";", offset);		// set index of end of cookie value
      		if (end == -1){
        		end = document.cookie.length;
			}
      		return unescape(document.cookie.substring(offset, end));
    	}
	}
}

function calcLatLng(updateAction)
{
	//set up bounding box
	var latlngPaddingIndex = adjustView ? 0 : map.getZoom();			
	var newBounds = map.getBounds();
	var ne = newBounds.getNorthEast();
	var sw = newBounds.getSouthWest();
	var maxLat = ne.lat() - latlngPadding[latlngPaddingIndex]; 
	var maxLng = ne.lng() - latlngPadding[latlngPaddingIndex];
	var minLat = sw.lat() + latlngPadding[latlngPaddingIndex];
	var minLng = sw.lng() + latlngPadding[latlngPaddingIndex];
	var searchBounds = 'MaxLat=' + maxLat + '&MaxLng=' + maxLng + '&MinLat=' + minLat + '&MinLng=' + minLng;
	
	//set up search criteria
	
	var SubCatId = document.getElementById("SubCatId").value;
	var regionId = document.getElementById("RegionId").value;
	//alert("SubCatId:"+SubCatId+"/"+"RegionId:"+regionId);
	//var SubCatId = document.getElementById("SubCatId").value;

	//var hPrice = document.getElementById("HPrice").value;
	//var lPrice = document.getElementById("LPrice").value;
	//var openHouse = document.getElementById("OpenFlag").checked ? 'Y' : 'N';
	
	//setup query string for xmllistings.asp
	//var qString = 'http://209.11.106.151/NYT_ULS_Redesign/CampsandSchools/XML_Listings.asp?';
	var qString = 'XML_Listings.asp?';
	var newPage = true;
	switch(updateAction)
	{	
		//send region id and search criteria
		case 'region':
			qString = qString +
			'RegionId=' + regionId +'&'+
			'Keyword=' + '&' +
			'SubCatId=' + SubCatId;
			break;
		
		//send coordinates and search criteria
		case 'coors':
			qString = qString +
			'lastPage=' + lastPage + '&' +'thisPage=' + thisPage + '&' +'RSCount=' + RSCount + '&' +'sLsIds'+ thisPage +'=' + boundsListings[thisPage];
			//'RegionId=' + regionId +'&'+
			//'Keyword=' + '&' +
			//'SubCatId=' + SubCatId;
			newPage = false;
			break;
			
		//send count, this and last pages, strLsIdsPriceDesc
		case 'page':
			qString = qString +
				'RSCount=' + RSCount + '&' +

			break;
	}
	populateMap(qString, newPage);
}

//create the marker and set up the event window
function createMarker(point,html,mIndex,mType,markerLetter) 
{
    var mLabel;
	var icon_off;
	var icon_on;
	
	var iconSingle = new GIcon();
	iconSingle.image = 'http://listings.nytimes.com/Redesign/images/map_markers/singlepoint_off.png';
	iconSingle.shadow = 'http://listings.nytimes.com/Redesign/images/map_markers/singlepoint_shadow.png';
	iconSingle.iconSize = new GSize(18.0, 25.0);
	iconSingle.shadowSize = new GSize(31.0, 25.0);
	iconSingle.iconAnchor = new GPoint(9.0, 12.0);
	iconSingle.infoWindowAnchor = new GPoint(9.0, 12.0);
	iconSingle.transparent = 'http://listings.nytimes.com/Redesign/images/map_markers/singlepoint_transparent.png';
	iconSingle.imageMap = [10,25,10,18,3,18,0,16,0,2,3,0,16,0,18,3,18,17,10,25];
	
	var iconMulti = new GIcon();
	iconMulti.image = 'http://listings.nytimes.com/Redesign/images/map_markers/multipoint_off.png';
	iconMulti.shadow = 'http://listings.nytimes.com/Redesign/images/map_markers/multipoint_shadow.png';
	iconMulti.transparent = 'http://listings.nytimes.com/Redesign/images/map_markers/multipoint_transparent.png';
	iconMulti.iconSize = new GSize(28.0, 33.0);
	iconMulti.shadowSize = new GSize(45.0, 33.0);
	iconMulti.iconAnchor = new GPoint(14.0, 16.0);
	iconMulti.infoWindowAnchor = new GPoint(14.0, 16.0);
	iconMulti.imageMap = [15,32,14,26,5,26,5,20,1,6,15,1,27,9,27,14,22,26,15,32];
	
	//assign number label and marker image
	switch(mType)
	{
		case 'single':
			mLabel = {"url":'http://listings.nytimes.com/Redesign/images/map_markers/' + markerLetter + '.png', "anchor":new GLatLng(2,3), "size":new GSize(14,14)};
			icon_off = new GIcon(iconSingle, markerOff[mType], mLabel);
			icon_on = new GIcon(iconSingle, markerOn[mType], mLabel);
			break;
		case 'multi':
			mLabel = {"url":'http://listings.nytimes.com/Redesign/images/map_markers/' + markerLetter + '.png', "anchor":new GLatLng(10,7), "size":new GSize(14,14)};
			icon_off = new GIcon(iconMulti, markerOff[mType], mLabel);
			icon_on = new GIcon(iconMulti, markerOn[mType], mLabel);
			break;
	}
	var marker= new GMarker(point,{icon:icon_off});
	
	//assign infowindow
	GEvent.addListener(marker, "click", function()
	{
		map.savePosition();
		marker.openInfoWindowHtml(html);
		//infoOpen = true;
	});
	
	GEvent.addListener(marker, "mouseover", function()
	{
		marker.setImage(markerOn[mType]);
	});
	
	GEvent.addListener(marker, "mouseout", function()
	{
		marker.setImage(markerOff[mType]);
	});
	return marker;	
}//createMarker()

// the loading screen
function showLoad()
{
	if(!loaded && !loading)
	{
		loading = true;
		function loadScreen(bounds, opt_weight, opt_color) 
		{
		  this.bounds_ = bounds;
		  this.weight_ = opt_weight || 2;
		  this.color_ = opt_color || "#888800";
		  this.opacity_ = .5;
		  this.background_ = "url('http://listings.nytimes.com/Redesign/images/map_markers/loading.gif') no-repeat center";
		}
		loadScreen.prototype = new GOverlay();
		
		loadScreen.prototype.initialize = function(map) 
		{
		  var div = document.createElement("div");
		  div.style.position = "absolute";
		  div.style.background = this.background_;
	
		  map.getPane(G_MAP_FLOAT_PANE).appendChild(div);
		
		  this.map_ = map;
		  this.div_ = div;
		  this.div_.opacity = this.opacity_;
		}
	
		loadScreen.prototype.remove = function() 
		{
		  this.div_.parentNode.removeChild(this.div_);
		}
		
		loadScreen.prototype.copy = function() 
		{
		  return new loadScreen(this.bounds_, this.weight_, this.color_, this.backgroundColor_, this.opacity_);
		}
		
		loadScreen.prototype.redraw = function(force) 
		{
		  // only need to redraw if the coordinate system has changed
		  if (!force) return;
		
		  var c1 = this.map_.fromLatLngToDivPixel(this.bounds_.getSouthWest());
		  var c2 = this.map_.fromLatLngToDivPixel(this.bounds_.getNorthEast());
		
		  this.div_.style.width = Math.abs(c2.x - c1.x) + "px";
		  this.div_.style.height = Math.abs(c2.y - c1.y) + "px";
		  this.div_.style.left = (Math.min(c2.x, c1.x) - this.weight_) + "px";
		  this.div_.style.top = (Math.min(c2.y, c1.y) - this.weight_) + "px";
		}
		
		var rBounds = map.getBounds();
		var southWest = rBounds.getSouthWest();
		var northEast = rBounds.getNorthEast();
		var rectBounds = new GLatLngBounds(
		    new GLatLng(southWest.lat(), southWest.lng()),
		    new GLatLng(northEast.lat(), northEast.lng())
			 );
		map.addOverlay(new loadScreen(rectBounds));
	}
}
